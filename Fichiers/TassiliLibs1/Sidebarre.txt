<template>
  <div>
    <div class="max-[799px]:hidden min-w-[240px] max-w-[240px] bg-black text-white h-full min-h-[100vh] p-[10px] pt-[5px]">

      <div class="p-[4px] bg-[blue] text-center border-[1px] font-bold border-white text-[22px] rounded-[2px] mb-[10px]">
        <Link :href="`/${page.props.tassiliSettings.tassiliPanel}`">DASHBOARD</Link>
      </div>

     <slot name="header"></slot>

      <Link
        v-for="(item, index) in tassiliroutes.routes"
        :key="index"
        :href="item.route"
        @click="tassililisting.resetActionIds()"
      >
        <div
          :class="currentPath.startsWith(item.route.split('?')[0])
            ? 'mt-[10px] flex mb-[10px] p-[5px] bg-[#8A2BE2] pl-[3px] pt-[10px] pb-[10px] text-[16px] rounded-r-[18px]'
            : 'pl-[3px] text-[16px] flex'"
        >
          <div class="pt-[2px]">
            <span class="material-icons" style="font-size: 18px;">
              {{ item.icon }} 
            </span>
          </div>
          <div class="pl-[5px]">
            <span>{{ item.label }}</span>
          </div>
        </div>
      </Link>



      <div class="pl-[3px]  text-[18px] flex">
         <div class="pt-[3px]" >
            <span class="material-icons text-[white]" style="font-size: 18px;">
                account_circle
               </span>
             
         </div>
         <div class="pl-[5px]">
         <span>{{ page.props.tassiliSettings.user.name }}</span>
        </div>
     </div>


       <div class="pl-[3px]  text-[16px] flex cursor-pointer">
         <div class="pt-[3px]" >
            <span class="material-icons text-[white]" style="font-size: 18px;" @click="logout()">
               settings_power
               </span>
             
         </div>
         <div class="pl-[5px]" @click="logout()">
         <span>Logout</span>
        </div>
     </div>



      <div class="pl-[3px]  text-[16px] cursor-pointer flex" 
      @click="resetSession()">
            <div class="pt-[2px]">
            <span class="material-icons" style="font-size: 18px;">
                settings
                </span>
            </div>
            <div class="pl-[5px]">
            <span>Clear Session</span>
           </div>
          </div>

        <slot name="footer"></slot> 


    </div>
  </div>
</template>

<script setup>
import { ref, computed } from 'vue';
import { Link, usePage } from '@inertiajs/vue3';
import { router } from '@inertiajs/vue3'
import { TassiliRoutes } from '@/Vendor/TassiliLibs/stores/tassiliRoutes';
import { TassiliListing } from '@/Vendor/TassiliLibs/stores/tassiliListing';

const tassiliroutes  = TassiliRoutes();
const tassililisting = TassiliListing();

const currentPath = computed(() => page.url.split('?')[0]);
const page = usePage();

const open = ref(false);

function logout() {
  if (confirm('Do you want to logout ?')) {
    router.post('/tassili/logout', {}, {
      preserveScroll: true,
    })
  }
}


function resetSession() {

  if (confirm('Do you want reset session ?')) {

    const panelStorage = 'tassili.' + page.props.tassiliSettings.tassiliPanel
    localStorage.removeItem(panelStorage)
    tassiliroutes.setRoutes(page.props.tassiliSettings.routes,page.props.tassiliSettings.tassiliPanel)
    const urlDirection = '/' + page.props.tassiliSettings.tassiliPanel
    router.get(urlDirection,{} ,{
      preserveScroll: false,
    })
  }

}



</script>
